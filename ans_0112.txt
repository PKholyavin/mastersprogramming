Длинное решение:
text_integer = []
for i in text:
    i = i.strip("\n")
    i = i.split(" ")
    for h in i:
        h = h.split("-")
        for l in h:
            l = int(l)
            text_integer.append(l)

text_integer = sorted(text_integer)
print(text_integer)
